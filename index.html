    String.prototype.toJadenCase = function () {
        let curentArr = [];
        let arrString = this.split(" ");
        arrString.forEach((item) => {
            let curentItem = item[0].toUpperCase() + item.slice(1);
            curentArr.push(curentItem)
        })
        return curentArr.join(" ")
    };
